handlers.GivePlayerCurrency = function GivePlayerCurrency(args)
{
    server.AddUserVirtualCurrency({PlayFabID: args.PlayerToGiveID, VirtualCurrency: args.CCode, Amount: args.AmountToGive});
    return 'Gave Currency';
}

handlers.BanPlayer = function (args, context) {
    var playerId = args.PlayFabId;
    var banReason = args.Reason || "Violation of Rules";
    var banDuration = args.Duration || 0; // 0 means permanent ban

    if (!playerId) {
        return { success: false, message: "Missing PlayFabId" };
    }

    var banRequest = {
        Bans: [{
            PlayFabId: playerId,
            BanReason: banReason,
            DurationInSeconds: banDuration
        }]
    };

    var banResult = server.BanUsers(banRequest);

    return { 
        success: true, 
        message: "Player banned successfully", 
        banInfo: banResult 
    };
};
